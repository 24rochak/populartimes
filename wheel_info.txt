
# cd into populartimes
# build wheel
python3 setup.py bdist_wheel

# create virtualenv
virtualenv --no-site-packages envname
source envname/bin/activate

# install package
pip3 install dist/populartimes-2.0-py3-none-any.whl

# check installation
pip3 show populartimes

# run
python3
import populartimes

# run with params.json
populartimes.main()

# also possible, with params as args
populartimes.execute("your_api_key", ["bar"], (48.132986, 11.566126), (48.142199, 11.580047), 20, 180)

# or simple run from init
python3 populartimes/__init__.py
